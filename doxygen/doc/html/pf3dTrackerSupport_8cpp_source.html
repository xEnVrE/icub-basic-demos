21  //I copied the transformation from the wikipedia. WARNING ??? !!!

24  U= -0.147F*float(R) -0.289F*float(G) +0.436F*float(B); //-255*436<U<255*436
25  U+=0.436F*255.0F; //0<U<255*2*0.436
26  U=U/(2.0F*0.436F);//0<U<255
